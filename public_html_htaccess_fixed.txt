<IfModule mod_rewrite.c>
  RewriteEngine On
  RewriteBase /

  # Secure backend access
  RewriteRule ^backend/(.*)$ - [F,L]
  
  # Forward /api requests to the backend public folder
  RewriteRule ^api/(.*)$ backend/public/index.php [L,QSA]

  # Forward download and signed routes to the backend
  RewriteRule ^(icard-download|joining-letter-download|verify|document-view|lead-document-view)/(.*)$ backend/public/index.php [L,QSA]

  # Forward /storage requests to the backend storage via the symlink
  # We use the symlink directly for better performance, but ensure it maps correctly
  # If 403 persists, we can use a rewrite to backend/public/index.php or similar, 
  # but standard symlink is better.
  # RewriteRule ^storage/(.*)$ backend/public/storage/$1 [L]

  # Standard React SPA routing
  RewriteRule ^index\.html$ - [L]
  RewriteCond %{REQUEST_FILENAME} !-f
  RewriteCond %{REQUEST_FILENAME} !-d
  RewriteCond %{REQUEST_URI} !^/api
  RewriteCond %{REQUEST_URI} !^/storage
  RewriteCond %{REQUEST_URI} !^/icard-download
  RewriteCond %{REQUEST_URI} !^/joining-letter-download
  RewriteCond %{REQUEST_URI} !^/document-view
  RewriteCond %{REQUEST_URI} !^/lead-document-view
  RewriteCond %{REQUEST_URI} !^/verify
  RewriteCond %{REQUEST_URI} !^/sitemap\.xml
  RewriteCond %{REQUEST_URI} !^/robots\.txt
  RewriteRule . /index.html [L]
</IfModule>

AddType application/manifest+json .webmanifest
AddType application/manifest+json .json
