<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /

    # 1. API and Storage Routes point to Laravel
    RewriteCond %{REQUEST_URI} ^/(api|sanctum|storage)/ [NC]
    RewriteRule ^(.*)$ backend/laravel/public/$1 [L]

    # 2. Redirect /assets requests directly to the dist folder
    # This avoids the MIME type error (text/html) on missing or misrouted assets
    RewriteRule ^assets/(.*)$ frontend/dist/assets/$1 [L]

    # 3. Handle specific root files in the dist folder (PWA, Service Worker, Root Assets)
    # Added workbox-*.js to resolve the Service Worker import error
    RewriteCond %{REQUEST_URI} ^/(manifest\.webmanifest|registerSW\.js|sw\.js|workbox-[a-z0-9]+\.js|robots\.txt|sitemap\.xml|favicon\.ico|logo\.webp|vite\.svg) [NC]
    RewriteRule ^(.*)$ frontend/dist/$1 [L]

    # 4. Everything else goes to React's index.html
    RewriteCond %{REQUEST_URI} !^/(api|sanctum|storage)/ [NC]
    RewriteCond %{REQUEST_URI} !^/frontend/dist/ [NC]
    RewriteRule ^(.*)$ frontend/dist/index.html [L]
</IfModule>
