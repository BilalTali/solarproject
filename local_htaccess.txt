<IfModule mod_rewrite.c>
    RewriteEngine On
    RewriteBase /

    # 1. API and Storage Routes point to Laravel
    RewriteCond %{REQUEST_URI} ^/(api|sanctum|storage)/ [NC]
    RewriteRule ^(.*)$ backend/laravel/public/$1 [L]

    # 2. Serve static assets if they exist in frontend/dist
    RewriteCond %{DOCUMENT_ROOT}/frontend/dist/$1 -f [OR]
    RewriteCond %{DOCUMENT_ROOT}/frontend/dist/$1 -d
    RewriteRule ^(.*)$ frontend/dist/$1 [L]

    # 3. Everything else goes to React's index.html
    RewriteCond %{REQUEST_URI} !^/(api|sanctum|storage)/ [NC]
    RewriteRule ^(.*)$ frontend/dist/index.html [L]
</IfModule>
